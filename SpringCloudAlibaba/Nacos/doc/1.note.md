nacos 控制台中的上线下线的功能指的是这个节点在配置中心中变成上线下线,并不会对节点本身造成影响,
下线后依旧是可以通过节点提供的接口访问,但是不能在通过服务注册中心用服务名称的方式访问
https://github.com/alibaba/nacos/issues/3877

```json
{
  "instanceList":[
    {
      "instanceId":"127.0.0.1#8080#DEFAULT#DEFAULT_GROUP@@Nacos-Spring-Boot",
      "ip":"127.0.0.1",
      "port":8080,
      "weight":1.0,
      "healthy":true,
      "enabled":false,
      "ephemeral":true,
      "clusterName":"DEFAULT",
      "serviceName":"DEFAULT_GROUP@@Nacos-Spring-Boot",

      },
      "lastBeat":1602756902401,
      "marked":false,
      "app":"DEFAULT",

      "instanceIdGenerator":"simple",
      "instanceHeartBeatTimeOut":15000,
      "ipDeleteTimeout":30000
    }]
}
```
